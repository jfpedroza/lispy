integer  : /-?[0-9]+/ ;
decimal  : /-?[0-9]+\.[0-9]+/ ;
number   : <decimal> | <integer>;
symbol   : /[a-zA-Z0-9_+\-*\/\^%\\=<>!&]+/ ;
string   : /"(\\.|[^"])*"/ ;
comment  : /;[^\r\n]*/ ;
sexpr    : '(' <expr>* ')' ;
qexpr    : '{' <expr>* '}' ;
expr     : <number> | <symbol> | <string> | <comment> | <sexpr> | <qexpr> ;
cname    : /\.[a-zA-Z0-9\-]+/ ;
lispy    : /^/ <cname>? <expr>* /$/ ;
