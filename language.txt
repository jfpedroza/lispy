
integer  : /-?[0-9]+/ ;
decimal  : /-?[0-9]+\.[0-9]+/ ;
number   : <decimal> | <integer>;
symbol   : '+' | '-' | '*' | '/' | '%' | '^' | "add" | "sub" | "mul" | "div" | "rem" | "pow" | "min" | "max" ;
sexpr    : '(' <expr>* ')' ;
expr     : <number> | <symbol> | <sexpr> ;
lispy    : /^/ <expr>* /$/ ;
