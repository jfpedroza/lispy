integer  : /-?[0-9]+/ ;
decimal  : /-?[0-9]+\.[0-9]+/ ;
number   : <decimal> | <integer>;
symbol   : '+' | '-' | '*' | '/' | '%' | '^' | "min" | "max" | "list" | "head" | "tail" | "join" | "eval" | "cons" | "len" | "init" ;
sexpr    : '(' <expr>* ')' ;
qexpr    : '{' <expr>* '}' ;
expr     : <number> | <symbol> | <sexpr> | <qexpr> ;
lispy    : /^/ <expr>* /$/ ;
